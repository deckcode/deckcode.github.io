    //    加载不完页面 driver卡死情况 自动设置线程停止加载事件
 public static class stoptLoad extends Thread{
     WebDriver driver = null;
     int sec = 0; 
     public stoptLoad(WebDriver driver,int sec){
         this.driver = driver;
         this.sec = sec;
     }

     @Override
     public void run() {
         try {
             Thread.sleep(sec*1000);
         } catch (InterruptedException e) {
             e.printStackTrace();
         }

         ((JavascriptExecutor) driver).executeScript("window.stop();");

     }
